    <form id="uploadForm">
        <input type="file" id="fileInput">
        <button type="button" onclick="generateLink()">Upload File</button>
    </form>
    <div id="shareLink">
        <p>Shareable Link:</p>
        <input type="text" id="fileLink" readonly>
    </div>
    <div id="fileList">
        <h2>Uploaded Files</h2>
        <ul id="files"></ul>
    </div>
    <script>
        let fileLinks = [];

        function generateLink() {
            const fileInput = document.getElementById('fileInput');
            const file = fileInput.files[0];
            if (file) {
                const link = URL.createObjectURL(file);
                fileLinks.push({ name: file.name, url: link });

                document.getElementById('fileLink').value = link;
                document.getElementById('shareLink').style.display = 'flex';
                addFileToList(file.name, link);
            } else {
                alert('Please select a file to upload.');
            }
        }

        function addFileToList(fileName, fileLink) {
            const fileList = document.getElementById('files');
            const listItem = document.createElement('li');
            listItem.innerHTML = `
                <strong>${fileName}</strong>
                <br>
                <a href="${fileLink}" target="_blank">View</a>
                |
                <a href="${fileLink}" download="${fileName}">Download</a>
            `;
            fileList.appendChild(listItem);
            document.getElementById('fileList').style.display = 'block';
        }
    </script>
